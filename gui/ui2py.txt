# In Anaconda or Command terminal, type:

pyuic5 project_window.ui -o ..\src\pzero\project_window_ui.py
pyuic5 base_view_window.ui -o ..\src\pzero\base_view_window_ui.py
pyuic5 import_window.ui -o ..\src\pzero\import_window_ui.py

pyuic5 project_window.ui > ../src/pzero/project_window_ui.py
pyuic5 base_view_window.ui > ../src/pzero/base_view_window_ui.py
pyuic5 import_window.ui > ../src/pzero/import_window_ui.py
